-- KPI Standardization & Automated Reporting Queries
-- Run against the kpi_reporting MySQL database
-- =============================================================

-- --------------------------------------------------------
-- 1. Monthly Revenue KPI by Region
-- --------------------------------------------------------
CREATE OR REPLACE VIEW vw_monthly_revenue AS
SELECT
    year_month,
    region,
    SUM(revenue)                          AS total_revenue,
    SUM(units_sold)                       AS total_units,
    ROUND(SUM(revenue) / SUM(units_sold), 2) AS avg_revenue_per_unit,
    COUNT(DISTINCT transaction_id)        AS transaction_count
FROM cleaned_data
GROUP BY year_month, region
ORDER BY year_month, region;


-- --------------------------------------------------------
-- 2. Department-Level KPI Summary
-- --------------------------------------------------------
CREATE OR REPLACE VIEW vw_dept_kpi AS
SELECT
    department,
    category,
    COUNT(transaction_id)                 AS total_transactions,
    SUM(revenue)                          AS total_revenue,
    ROUND(AVG(revenue), 2)               AS avg_revenue,
    ROUND(AVG(revenue_per_unit), 2)      AS avg_rev_per_unit
FROM cleaned_data
GROUP BY department, category
ORDER BY total_revenue DESC;


-- --------------------------------------------------------
-- 3. Top Performing Regions (Ranked)
-- --------------------------------------------------------
CREATE OR REPLACE VIEW vw_region_ranking AS
SELECT
    region,
    SUM(revenue)                          AS total_revenue,
    RANK() OVER (ORDER BY SUM(revenue) DESC) AS revenue_rank
FROM cleaned_data
GROUP BY region;


-- --------------------------------------------------------
-- 4. Month-over-Month Revenue Growth
-- --------------------------------------------------------
CREATE OR REPLACE VIEW vw_mom_growth AS
SELECT
    year_month,
    SUM(revenue)                          AS monthly_revenue,
    LAG(SUM(revenue)) OVER (ORDER BY year_month) AS prev_month_revenue,
    ROUND(
        (SUM(revenue) - LAG(SUM(revenue)) OVER (ORDER BY year_month))
        / NULLIF(LAG(SUM(revenue)) OVER (ORDER BY year_month), 0) * 100,
        2
    )                                     AS mom_growth_pct
FROM cleaned_data
GROUP BY year_month;


-- --------------------------------------------------------
-- 5. Pipeline Run Log Report
-- --------------------------------------------------------
SELECT
    run_timestamp,
    source_file,
    rows_loaded,
    status
FROM reporting_log
ORDER BY run_timestamp DESC
LIMIT 20;
